"use client";

import { useState, useEffect } from "react";
import { useForm } from "react-hook-form";
import { useRouter } from "next/navigation";
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";
import { Label } from "@/components/ui/label";
import { RadioGroup, RadioGroupItem } from "@/components/ui/radio-group";
import {
  Select,
  SelectContent,
  SelectItem,
  SelectTrigger,
  SelectValue,
} from "@/components/ui/select";
import { toast } from "sonner";
import { apiClient } from "@/lib/api-client";

interface SaleFormData {
  customerName: string;
  amount: number;
  vatOption: "included" | "separate" | "none";
  paymentMethod: "CASH" | "BANK" | "WALLET" | "CREDIT";
  paymentAccountCode: string;
  gateway?: "ESEWA" | "KHALTI";
  description: string;
}

export function SaleForm({ onSuccess }: { onSuccess?: () => void }) {
  const router = useRouter();
  const [isSubmitting, setIsSubmitting] = useState(false);
  const [accounts, setAccounts] = useState<any[]>([]);
  const { register, handleSubmit, watch, setValue, formState: { errors } } = useForm<SaleFormData>({
    defaultValues: {
      vatOption: "included",
      paymentMethod: "CASH",
      paymentAccountCode: "1-1001",
      customerName: "",
      description: "",
    },
  });

  const amount = watch("amount") || 0;
  const vatOption = watch("vatOption");
  const paymentMethod = watch("paymentMethod");
  const selectedGateway = watch("gateway");

  // Fetch accounts on mount
  useEffect(() => {
    loadAccounts();
  }, []);

  const loadAccounts = async () => {
    try {
      const data = await apiClient("/accounts");
      setAccounts(data);
    } catch (error) {
      console.error("Failed to load accounts:", error);
    }
  };

  // Calculate VAT
  const calculateVAT = () => {
    if (vatOption === "none" || amount === 0) return 0;
    
    if (vatOption === "included") {
      return (amount * 0.13) / 1.13;
    } else {
      return amount * 0.13;
    }
  };

  const calculateTotal = () => {
    if (vatOption === "included") {
      return amount; // VAT already included in amount
    } else if (vatOption === "separate") {
      return amount + calculateVAT(); // Add VAT on top
    } else {
      return amount; // No VAT
    }
  };

  const vatAmount = calculateVAT();
  const netAmount = vatOption === "included" ? amount - vatAmount : amount;
  const totalAmount = calculateTotal();

  // Get payment accounts based on method
  const getPaymentAccounts = () => {
    switch (paymentMethod) {
      case "CASH":
        return accounts.filter(a => a.code === "1-1001");
      case "BANK":
        return accounts.filter(a => a.code.startsWith("1-1002") || a.code.startsWith("1-1003"));
      case "WALLET":
        return accounts.filter(a => a.code.startsWith("1-1004") || a.code.startsWith("1-1005"));
      default:
        return [];
    }
  };

  const onSubmit = async (data: SaleFormData) => {
    if (data.amount <= 0) {
      toast.error("Please enter a valid amount");
      return;
    }

    setIsSubmitting(true);
    try {
      // If it's a wallet payment with gateway selected, use gateway sale endpoint
      if (data.paymentMethod === "WALLET" && data.gateway) {
        await apiClient("/transactions/sale/gateway", {
          method: "POST",
          body: JSON.stringify({
            customerName: data.customerName || "Walk-in Customer",
            amount: data.amount,
            vatIncluded: data.vatOption === "included",
            gateway: data.gateway,
            description: data.description || `Sale via ${data.gateway}`,
          }),
        });
      } else if (data.paymentMethod === "CREDIT") {
        // Credit sale
        await apiClient("/transactions/sale/credit", {
          method: "POST",
          body: JSON.stringify({
            customerName: data.customerName,
            amount: data.amount,
            vatIncluded: data.vatOption === "included",
            description: data.description || "Credit Sale",
          }),
        });
      } else {
        // Cash/Bank/Wallet sale
        await apiClient("/transactions/sale", {
          method: "POST",
          body: JSON.stringify({
            customerName: data.customerName || "Walk-in Customer",
            amount: data.amount,
            vatIncluded: data.vatOption === "included",
            paymentMethod: data.paymentMethod,
            paymentAccountCode: data.paymentAccountCode,
            description: data.description || "Sale of goods",
          }),
        });
      }

      toast.success("✅ Sale recorded successfully!");
      
      if (onSuccess) {
        onSuccess();
      } else {
        router.push("/sales");
      }
    } catch (error: any) {
      console.error("Sale failed:", error);
      toast.error(error.message || "❌ Failed to record sale. Please try again.");
    } finally {
      setIsSubmitting(false);
    }
  };

  return (
    <Card className="max-w-2xl mx-auto">
      <CardHeader>
        <CardTitle className="text-2xl flex items-center gap-2">
          <span className="text-green-600">💰</span> New Sale (Money In)
        </CardTitle>
        <p className="text-sm text-gray-500">Record a new sale transaction</p>
      </CardHeader>
      <CardContent>
        <form onSubmit={handleSubmit(onSubmit)} className="space-y-6">
          
          {/* ============ CUSTOMER SECTION ============ */}
          <div className="space-y-2">
            <Label className="text-base">👤 Customer</Label>
            <Input
              {...register("customerName")}
              placeholder="Customer name (leave empty for walk-in)"
              className="h-11"
            />
          </div>

          {/* ============ AMOUNT SECTION ============ */}
          <div className="space-y-2">
            <Label className="text-base">💵 Amount (रू)</Label>
            <Input
              type="number"
              step="0.01"
              min="0"
              {...register("amount", { 
                required: "Amount is required",
                min: { value: 0, message: "Amount must be positive" },
                valueAsNumber: true,
              })}
              placeholder="Enter sale amount"
              className="h-14 text-2xl font-bold text-center"
            />
            {errors.amount && (
              <p className="text-red-500 text-sm">{errors.amount.message}</p>
            )}
          </div>

          {/* ============ VAT SECTION ============ */}
          <div className="space-y-3">
            <Label className="text-base">🧾 VAT Option</Label>
            <RadioGroup
              defaultValue="included"
              onValueChange={(value) => setValue("vatOption", value as any)}
              className="grid grid-cols-3 gap-3"
            >
              <div className={`flex items-center space-x-2 border rounded-lg p-4 cursor-pointer hover:bg-gray-50 ${vatOption === "included" ? "border-blue-500 bg-blue-50" : ""}`}>
                <RadioGroupItem value="included" id="vat-included" />
                <Label htmlFor="vat-included" className="cursor-pointer">
                  <span className="font-medium">VAT Included</span>
                  <br />
                  <span className="text-xs text-gray-500">Amount मा VAT समावेश छ</span>
                </Label>
              </div>
              
              <div className={`flex items-center space-x-2 border rounded-lg p-4 cursor-pointer hover:bg-gray-50 ${vatOption === "separate" ? "border-blue-500 bg-blue-50" : ""}`}>
                <RadioGroupItem value="separate" id="vat-separate" />
                <Label htmlFor="vat-separate" className="cursor-pointer">
                  <span className="font-medium">Add VAT</span>
                  <br />
                  <span className="text-xs text-gray-500">Amount माथि 13% VAT जोड्ने</span>
                </Label>
              </div>
              
              <div className={`flex items-center space-x-2 border rounded-lg p-4 cursor-pointer hover:bg-gray-50 ${vatOption === "none" ? "border-blue-500 bg-blue-50" : ""}`}>
                <RadioGroupItem value="none" id="vat-none" />
                <Label htmlFor="vat-none" className="cursor-pointer">
                  <span className="font-medium">No VAT</span>
                  <br />
                  <span className="text-xs text-gray-500">VAT लागु हुदैन</span>
                </Label>
              </div>
            </RadioGroup>
          </div>

          {/* ============ PAYMENT METHOD SECTION ============ */}
          <div className="space-y-3">
            <Label className="text-base">💳 Payment Method</Label>
            <RadioGroup
              defaultValue="CASH"
              onValueChange={(value) => {
                setValue("paymentMethod", value as any);
                // Set default account based on method
                if (value === "CASH") setValue("paymentAccountCode", "1-1001");
                if (value === "BANK") setValue("paymentAccountCode", "1-1002");
                if (value === "WALLET") setValue("paymentAccountCode", "1-1004");
              }}
              className="grid grid-cols-2 lg:grid-cols-4 gap-3"
            >
              {[
                { value: "CASH", label: "💰 Cash", icon: "💰" },
                { value: "BANK", label: "🏦 Bank", icon: "🏦" },
                { value: "WALLET", label: "📱 Wallet", icon: "📱" },
                { value: "CREDIT", label: "📋 Credit", icon: "📋" },
              ].map((pm) => (
                <div
                  key={pm.value}
                  className={`flex flex-col items-center justify-center space-y-2 border rounded-lg p-4 cursor-pointer hover:bg-gray-50 ${
                    paymentMethod === pm.value ? "border-blue-500 bg-blue-50" : ""
                  }`}
                  onClick={() => {
                    setValue("paymentMethod", pm.value as any);
                  }}
                >
                  <RadioGroupItem value={pm.value} id={`pm-${pm.value}`} className="sr-only" />
                  <span className="text-2xl">{pm.icon}</span>
                  <span className="font-medium text-sm">{pm.label}</span>
                </div>
              ))}
            </RadioGroup>
          </div>

          {/* ============ WALLET SPECIFIC (Gateway Selection) ============ */}
          {paymentMethod === "WALLET" && (
            <div className="space-y-3 p-4 bg-blue-50 rounded-lg">
              <Label className="text-base">📱 Select Wallet / Gateway</Label>
              <div className="grid grid-cols-2 gap-3">
                {[
                  { value: "ESEWA", label: "eSewa", color: "bg-green-100" },
                  { value: "KHALTI", label: "Khalti", color: "bg-purple-100" },
                ].map((gw) => (
                  <div
                    key={gw.value}
                    className={`${gw.color} p-4 rounded-lg cursor-pointer border-2 ${
                      selectedGateway === gw.value ? "border-blue-500" : "border-transparent"
                    }`}
                    onClick={() => setValue("gateway", gw.value as any)}
                  >
                    <p className="font-bold text-center">{gw.label}</p>
                    <p className="text-xs text-center mt-1">1.5% gateway fee</p>
                  </div>
                ))}
              </div>
            </div>
          )}

          {/* ============ ACCOUNT SELECTION (for non-credit) ============ */}
          {paymentMethod !== "CREDIT" && paymentMethod !== "WALLET" && (
            <div className="space-y-2">
              <Label className="text-base">🏦 Select {paymentMethod} Account</Label>
              <Select
                defaultValue="1-1001"
                onValueChange={(value) => setValue("paymentAccountCode", value)}
              >
                <SelectTrigger className="w-full h-11">
                  <SelectValue placeholder="Select account" />
                </SelectTrigger>
                <SelectContent>
                  {getPaymentAccounts().map((account) => (
                    <SelectItem key={account.id} value={account.code}>
                      {account.name} (Rs. {Number(account.currentBalance).toLocaleString()})
                    </SelectItem>
                  ))}
                </SelectContent>
              </Select>
            </div>
          )}

          {/* ============ SUMMARY SECTION ============ */}
          {amount > 0 && (
            <div className="bg-gradient-to-r from-gray-50 to-blue-50 p-4 rounded-lg border space-y-3">
              <h4 className="font-bold text-lg text-gray-700">📊 Transaction Summary</h4>
              
              <div className="space-y-2 text-sm">
                <div className="flex justify-between">
                  <span className="text-gray-600">Sale Amount (Net)</span>
                  <span className="font-medium">रू {netAmount.toLocaleString("en-IN", { minimumFractionDigits: 2 })}</span>
                </div>
                
                {vatAmount > 0 && (
                  <div className="flex justify-between">
                    <span className="text-gray-600">VAT (13%)</span>
                    <span className="font-medium text-orange-600">+ रू {vatAmount.toLocaleString("en-IN", { minimumFractionDigits: 2 })}</span>
                  </div>
                )}
                
                <div className="border-t pt-2 flex justify-between text-base">
                  <span className="font-bold">Total Received</span>
                  <span className={`font-bold text-lg ${paymentMethod === "CREDIT" ? "text-red-600" : "text-green-600"}`}>
                    रू {totalAmount.toLocaleString("en-IN", { minimumFractionDigits: 2 })}
                  </span>
                </div>

                {paymentMethod === "WALLET" && selectedGateway && (
                  <div className="border-t pt-2 text-xs text-gray-500">
                    <p>⚠️ Gateway fee (1.5%) will be deducted at settlement</p>
                    <p>Net settlement: रू {(totalAmount * 0.985).toLocaleString("en-IN", { minimumFractionDigits: 2 })}</p>
                  </div>
                )}
              </div>
            </div>
          )}

          {/* ============ DESCRIPTION ============ */}
          <div className="space-y-2">
            <Label className="text-base">📝 Note / Description</Label>
            <Input
              {...register("description")}
              placeholder="e.g., Sold T-shirt to Ram, Invoice #123"
              className="h-11"
            />
          </div>

          {/* ============ SUBMIT BUTTON ============ */}
          <Button
            type="submit"
            className="w-full h-14 text-lg font-bold"
            disabled={isSubmitting || amount <= 0}
          >
            {isSubmitting ? (
              <span className="flex items-center gap-2">
                <span className="animate-spin">⏳</span> Recording Sale...
              </span>
            ) : (
              <span className="flex items-center gap-2">
                ✅ Record Sale - रू {totalAmount.toLocaleString("en-IN", { minimumFractionDigits: 2 })}
              </span>
            )}
          </Button>
        </form>
      </CardContent>
    </Card>
  );
}
